317 /* zv_star -- computes componentwise (Hadamard) product of x1 and x2
318  -- result out is returned */
319 ZVEC *zv_star(x1, x2, out)
320 ZVEC *x1, *x2, *out;
321 {
322  int i;
323  Real t_re, t_im;
324

330

334  t_re = x1->ve[i].re*x2->ve[i].re - x1->ve[i].im*x2->ve[i].im;
