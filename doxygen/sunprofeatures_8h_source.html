56 // The basic idiom is:
57 // #ifndef R123_SOMETHING
58 // #if some condition
59 // #define R123_SOMETHING 1
60 // #else
61 // #define R123_SOMETHING 0
62 // #endif
63 // #endif
64 // This idiom allows an external user to override any decision
65 // in this file with a command-line -DR123_SOMETHING=1 or -DR123_SOMETHINE=0
66 
67 // An alternative idiom is:
68 // #ifndef R123_SOMETHING
69 // #define R123_SOMETHING (some boolean expression)
70 // #endif
71 // where the boolean expression might contain previously-defined R123_SOMETHING_ELSE
72 // pp-symbols.
