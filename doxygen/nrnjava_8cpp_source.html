363 static char** js2charpool(jstring js) {
364 #define imax 5
365  static char** cs;
366  static int i;
367  // allow up to 5 calls before freeing
368  // the problem is that several of these can be put on the
369  // hoc stack before an early one is copied into strdef
370  // and until then the early one must stay in existence
371  if (!cs) {
372  cs = new char*[imax];
373  for (i=0; i < imax; ++i) {
374  cs[i] = nil;
375  }
376  i = 0;
377  }
378  const char* jc = nrnjava_env->GetStringUTFChars(js, 0);
379  if (cs[i]) {
380  delete [] cs[i];
381  cs[i] = nil;
382  }
383  i = (i+1)%imax;
384  cs[i] = new char[strlen(jc) + 1];
385  strcpy(cs[i], jc);
386  nrnjava_env->ReleaseStringUTFChars(js, jc);
387  return cs + i;
388 }

407 Object** nj_j2hObject(jobject jo, int type) {
408  Object** po = nil;
409  if (jo == 0) { //null

411  return po;
412  }

414  void* v = nrnjava_env->NewGlobalRef(jo);
415  Symbol* tsym = nrn_jclass_symlist->item(type);
416  po = hoc_temp_objvar(tsym, v);
417  return po;
418  }else{
419  void* v;
420  Object* o;
421  switch (type) {

442 // printf("java2nrn_ometh invoking %s.%s\n", hoc_object_name(ho), method->name);
443  // I don't know how to return two things at once so we call two
444  // methods. The first returns a java object and the second tells
445  // the type.
446  //type 0+ is a registered java object and is the template id.
447  //type -1 is an unregistered java object -- enclose in JavaObject
448  //type -2 is a HocObject and the jobject has a field called
449  // long hocObjectCast which we can cast to the correct object.
450  // other types are for Java objects that extend HocObject
451  // but all have a field called "long hocObjectCast"
452  Object** po = nil;
453  if (method->s_varn) {overloaded(ho, method);}
454 
455  jobject jo = nrnjava_env->CallStaticObjectMethod(
456  neuronCls, invokeObjectMethodID,
457  (jobject)ho->u.this_pointer,
458  -(jint)ho->ctemplate->id - 1,
459  (jint)method->u.u_auto, (jint)method->s_varn);
460  if (nrnjava_env->ExceptionOccurred()) {
461  nrnjava_env->ExceptionDescribe();
462  hoc_execerror("Java Exception for", method->name);
463  }
464  jint type = 0;
465  if (jo) {
466  type = nrnjava_env->CallStaticIntMethod(
