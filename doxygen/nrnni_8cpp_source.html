26  // nilsdev --verbose

28  // at cat /proc/pci and from the Bus - device section the
29  // devicePCI_ID is the 8 digit number after PCI device.

31 //printf("trying devicePCI_ID = 0x%x\n", devicePCI_ID);

62 void NrnDAQ::initMite() {
63  tAddressSpace bar0;
64  u32 physicalBar1;
65 
66  //Skip MITE initialization for PCMCIA boards
67  //(which do not have a MITE DMA controller)
68  if(!bus->get(kIsPciPxiBus,0)) return;
69 
70  bar0 = bus->createAddressSpace(kPCI_BAR0);
71 
72  //Get the physical address of the DAQ board
73  physicalBar1 = bus->get(kBusAddressPhysical,kPCI_BAR1);
74  //Tell the MITE to enable BAR1, where the rest of the board's regist
75  bar0.write32(0xC0, (physicalBar1 & 0xffffff00L) | 0x80);
76 
77  bus->destroyAddressSpace(bar0);
78 }
