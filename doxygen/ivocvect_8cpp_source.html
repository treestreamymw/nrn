3817 static inline void SWAP(int* A, int* B)
3818 {
3819  int tmp = *A; *A = *B; *B = tmp;
3820 }
3821 
3822 /* This should be replaced by a standard ANSI macro. */
3823 #define BYTES_PER_WORD 8
3824 #define BYTES_PER_LONG 4
3825 
3826 /* The next 4 #defines implement a very fast in-line stack abstraction. */
3827 
3828 #define STACK_SIZE (BYTES_PER_WORD * BYTES_PER_LONG)
3829 #define PUSH(LOW,HIGH) do {top->lo = LOW;top++->hi = HIGH;} while (0)
3830 #define POP(LOW,HIGH) do {LOW = (--top)->lo;HIGH = top->hi;} while (0)
3831 #define STACK_NOT_EMPTY (stack < top)
3832 
3833 /* Discontinue quicksort algorithm when partition gets below this size.
3834  This particular magic number was chosen to work best on a Sun 4/260. */
3835 #define MAX_THRESH 4
3836 
3837 
3838 /* Order size using quicksort. This implementation incorporates
3839  four optimizations discussed in Sedgewick:
3840 
3841  1. Non-recursive, using an explicit stack of pointer that
3842  store the next array partition to sort. To save time, this
3843  maximum amount of space required to store an array of
3844  MAX_INT is allocated on the stack. Assuming a 32-bit integer,
3845  this needs only 32 * sizeof (stack_node) == 136 bits. Pretty
3846  cheap, actually.
3847 
3848  2. Chose the pivot element using a median-of-three decision tree.
3849  This reduces the probability of selecting a bad pivot value and
3850  eliminates certain extraneous comparisons.
3851 
3852  3. Only quicksorts TOTAL_ELEMS / MAX_THRESH partitions, leaving
3853  insertion sort to order the MAX_THRESH items within each partition.
3854  This is a big win, since insertion sort is faster for small, mostly
3855  sorted array segements.
3856 
3857  4. The larger of the two sub-partitions is always pushed onto the
3858  stack first, with the algorithm then concentrating on the
3859  smaller partition. This *guarantees* no more than log (n)
3860  stack size is needed! */
3861 
3862 int nrn_mlh_gsort (double* vec, int *base_ptr, int total_elems, doubleComparator cmp)
3863 {
3864 /* Stack node declarations used to store unfulfilled partition obligations. */
3865  struct stack_node { int *lo; int *hi; };
3866  int pivot_buffer;
3867  int max_thresh = MAX_THRESH;
3868 
3869  if (total_elems > MAX_THRESH)
3870  {
3871  int *lo = base_ptr;
3872  int *hi = lo + (total_elems - 1);
3873  int *left_ptr;
3874  int *right_ptr;
3875  stack_node stack[STACK_SIZE]; /* Largest size needed for 32-bit int!!! */
3876  stack_node *top = stack + 1;
3877 
3878  while (STACK_NOT_EMPTY)
3879  {
3880  {
3881  int *pivot = &pivot_buffer;
3882  {
3883  /* Select median value from among LO, MID, and HI. Rearrange
3884  LO and HI so the three values are sorted. This lowers the
3885  probability of picking a pathological pivot value and
3886  skips a comparison for both the LEFT_PTR and RIGHT_PTR. */
3887 
3888  int *mid = lo + ((hi - lo) >> 1);
3889 
3890  if ((*cmp) (vec[*mid], vec[*lo]) < 0)
3891  SWAP (mid, lo);
3892  if ((*cmp) (vec[*hi], vec[*mid]) < 0)
3893  {
3894  SWAP (mid, hi);
3895  if ((*cmp) (vec[*mid], vec[*lo]) < 0)
3896  SWAP (mid, lo);
3897  }
3898  *pivot = *mid;
3899  pivot = &pivot_buffer;
3900  }
3901  left_ptr = lo + 1;
3902  right_ptr = hi - 1;
