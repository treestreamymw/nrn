3  * sptree.c: The following code implements the basic operations on
4  * an event-set or priority-queue implemented using splay trees:
5  *
6 Hines changed to void spinit(SPTREE**) for use with TQueue.
7  * SPTREE *spinit( compare ) Make a new tree
8  * int spempty(); Is tree empty?
9  * SPBLK *spenq( n, q ) Insert n in q after all equal keys.
10  * SPBLK *spdeq( np ) Return first key under *np, removing it.
11  * SPBLK *spenqprior( n, q ) Insert n in q before all equal keys.
12  * void splay( n, q ) n (already in q) becomes the root.
13  *
14  * In the above, n points to an SPBLK type, while q points to an
15  * SPTREE.
16  *
17  * The implementation used here is based on the implementation
18  * which was used in the tests of splay trees reported in:
19  *
20  * An Empirical Comparison of Priority-Queue and Event-Set Implementations,
21  * by Douglas W. Jones, Comm. ACM 29, 4 (Apr. 1986) 300-311.
22  *
23  * The changes made include the addition of the enqprior
24  * operation and the addition of up-links to allow for the splay
25  * operation. The basic splay tree algorithms were originally
26  * presented in:
27  *
28  * Self Adjusting Binary Trees,
29  * by D. D. Sleator and R. E. Tarjan,
30  * Proc. ACM SIGACT Symposium on Theory
31  * of Computing (Boston, Apr 1983) 235-245.
32  *
33  * The enq and enqprior routines use variations on the
34  * top-down splay operation, while the splay routine is bottom-up.
35  * All are coded for speed.
36  *
37  * Written by:
38  * Douglas W. Jones
39  *
40  * Translated to C by:
41  * David Brower, daveb@rtech.uucp
42  *
43  * Thu Oct 6 12:11:33 PDT 1988 (daveb) Fixed spdeq, which was broken
44  * handling one-node trees. I botched the pascal translation of
45  * a VAR parameter.
46  */

85  * spenq() -- insert item in a tree.
86  *
87  * put n in q after all other nodes with the same key; when this is
88  * done, n will be the root of the splay tree representing q, all nodes
89  * in q with keys less than or equal to that of n will be in the
90  * left subtree, all with greater keys will be in the right subtree;
91  * the tree is split into these subtrees from the top down, with rotations
92  * performed along the way to shorten the left branch of the right subtree
93  * and the right branch of the left subtree
94  */

231  * spdeq() -- return and remove head node from a subtree.
232  *
233  * remove and return the head node from the node set; this deletes
234  * (and returns) the leftmost node from q, replacing it with its right
235  * subtree (if there is one); on the way to the leftmost node, rotations
236  * are performed to shorten the left branch of the tree
237  */

239 spdeq( SPBLK** np ) /* pointer to a node pointer */
240 
241 {

243  register SPBLK * next; /* the next thing to deal with */
244  register SPBLK * left; /* the left child of next */
245  register SPBLK * farleft; /* the left child of left */
246  register SPBLK * farfarleft; /* the left child of farleft */
247 
248  if( np == NULL || *np == NULL )
249  {
250  deq = NULL;
251  }
252  else
253  {
254  next = *np;
255  left = next->leftlink;
256  if( left == NULL )
257  {
258  deq = next;
259  *np = next->rightlink;
260 
261  if( *np != NULL )
262  (*np)->uplink = NULL;
263 
264  }
265  else for(;;) /* left is not null */
266  {
267  /* next is not it, left is not NULL, might be it */
268  farleft = left->leftlink;
269  if( farleft == NULL )
270  {
271  deq = left;
272  next->leftlink = left->rightlink;
273  if( left->rightlink != NULL )
274  left->rightlink->uplink = next;
275  break;
276  }
277 
278  /* next, left are not it, farleft is not NULL, might be it */
279  farfarleft = farleft->leftlink;
280  if( farfarleft == NULL )
281  {
282  deq = farleft;
283  left->leftlink = farleft->rightlink;
284  if( farleft->rightlink != NULL )
285  farleft->rightlink->uplink = left;
286  break;
287  }
288 
289  /* next, left, farleft are not it, rotate */
290  next->leftlink = farleft;
291  farleft->uplink = next;
292  left->leftlink = farleft->rightlink;
293  if( farleft->rightlink != NULL )
294  farleft->rightlink->uplink = left;
295  farleft->rightlink = left;
296  left->uplink = farleft;
297  next = farleft;
298  left = farfarleft;
299  }

441  *
442  * splay() -- reorganize the tree.
443  *
444  * the tree is reorganized so that n is the root of the
445  * splay tree representing q; results are unpredictable if n is not
446  * in q to start with; q is split from n up to the old root, with all
447  * nodes to the left of n ending up in the left subtree, and all nodes
448  * to the right of n ending up in the right subtree; the left branch of
449  * the right subtree and the right branch of the left subtree are
450  * shortened in the process
451  *
452  * this code assumes that n is not NULL and is in q; it can sometimes
453  * detect n not in q and complain
454  */
455 
456 void
457 splay( SPBLK* n, SPTREE* q )
458 {
459  register SPBLK * up; /* points to the node being dealt with */
460  register SPBLK * prev; /* a descendent of up, already dealt with */
461  register SPBLK * upup; /* the parent of up */
462  register SPBLK * upupup; /* the grandparent of up */
463  register SPBLK * left; /* the top of left subtree being built */
464  register SPBLK * right; /* the top of right subtree being built */
465 
466 #if BBTQ != 4 && BBTQ != 5
467  n->cnt++; /* bump reference count */
468 #endif
469 
470  left = n->leftlink;
471  right = n->rightlink;
472  prev = n;
473  up = prev->uplink;
474 
475  q->splays++;
476 
477  while( up != NULL )
478  {
479  q->splayloops++;
480 
481  /* walk up the tree towards the root, splaying all to the left of
482  n into the left subtree, all to right into the right subtree */
483 
484  upup = up->uplink;
485  if( up->leftlink == prev ) /* up is to the right of n */
486  {
487  if( upup != NULL && upup->leftlink == up ) /* rotate */
488  {
489  upupup = upup->uplink;
490  upup->leftlink = up->rightlink;
491  if( upup->leftlink != NULL )
492  upup->leftlink->uplink = upup;
493  up->rightlink = upup;
494  upup->uplink = up;
495  if( upupup == NULL )
496  q->root = up;
497  else if( upupup->leftlink == upup )
498  upupup->leftlink = up;
499  else
500  upupup->rightlink = up;
501  up->uplink = upupup;
502  upup = upupup;
503  }
504  up->leftlink = right;
505  if( right != NULL )
506  right->uplink = up;
507  right = up;
508 
509  }
510  else /* up is to the left of n */
511  {
512  if( upup != NULL && upup->rightlink == up ) /* rotate */
513  {
514  upupup = upup->uplink;
515  upup->rightlink = up->leftlink;
516  if( upup->rightlink != NULL )
517  upup->rightlink->uplink = upup;
518  up->leftlink = upup;
519  upup->uplink = up;
520  if( upupup == NULL )
521  q->root = up;
522  else if( upupup->rightlink == upup )
523  upupup->rightlink = up;
524  else
525  upupup->leftlink = up;
526  up->uplink = upupup;
527  upup = upupup;
528  }
529  up->rightlink = left;
530  if( left != NULL )
531  left->uplink = up;
532  left = up;
533  }
534  prev = up;
535  up = upup;
536  }
537 
538 # ifdef DEBUG
539  if( q->root != prev )
540  {
541 /* fprintf(stderr, " *** bug in splay: n not in q *** " ); */
542  abort();
543  }
544 # endif
545 
546  n->leftlink = left;
547  n->rightlink = right;
548  if( left != NULL )
549  left->uplink = n;
550  if( right != NULL )
551  right->uplink = n;
552  q->root = n;
553  n->uplink = NULL;
554 
555 } /* splay */
