549  treesolver
550 
551  a - above the diagonal
552  b - below the diagonal
553  c - used to define diagonal: diag = c + dt * dbase
554  dbase - together with c, used to define diagonal
555  rhs - right hand side, which is changed to the result
556  pindex - parent indices
557  n - number of states
558  */
559  long i, pin;
560  double* d = (double*)malloc(sizeof(double) * n);

564

566

568  myd = d;
569  mydbase = dbase;
570  myc = c;
571  for (i = 0; i < n; i++) {
572  *myd++ = *myc++ + dt * (*mydbase++);
573  }
574 
575  /* triang */
576  for (i = n - 1; i > 0; --i) {
577  pin = pindex[i];
578  if (pin > -1) {
579  double p;
580  p = dt * a[i] / d[i];
581  d[pin] -= dt * p * b[i];
582  rhs[pin] -= p * rhs[i];
583  }
584  }

587  pin = pindex[i];
588  if (pin > -1) {
589  rhs[i] -= dt * b[i] * rhs[pin];
590  }
591  rhs[i] /= d[i];
592  }
593
