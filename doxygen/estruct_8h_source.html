171 /*
172  * There is a window structure allocated for every active display window. The
173  * windows are kept in a big list, in top to bottom screen order, with the
174  * listhead at "wheadp". Each window contains its own values of dot and mark.
175  * The flag field contains some bits that are set by commands to guide
176  * redisplay; although this is a bit of a compromise in terms of decoupling,
177  * the full blown redisplay is just too expensive to run for every input
178  * character.
179  */
180 typedef struct WINDOW {
181  struct WINDOW *w_wndp; /* Next window */
182  struct BUFFER *w_bufp; /* Buffer displayed in window */
183  struct LINE *w_linep; /* Top line in the window */
184  struct LINE *w_dotp; /* Line containing "." */
185  struct LINE *w_markp; /* Line containing "mark" */
186  int w_doto; /* Byte offset for "." */
187  int w_marko; /* Byte offset for "mark" */
188  char w_toprow; /* Origin 0 top row of window */
189  char w_ntrows; /* # of rows of text in window */
190  char w_force; /* If NZ, forcing row. */
191  char w_flag; /* Flags. */
192 } WINDOW;

200 /*
201  * Text is kept in buffers. A buffer header, described below, exists for every
202  * buffer in the system. The buffers are kept in a big list, so that commands
203  * that search for a buffer by name can find the buffer header. There is a
204  * safe store for the dot and mark in the header, but this is only valid if
205  * the buffer is not being displayed (that is, if "b_nwnd" is 0). The text for
206  * the buffer is kept in a circularly linked list of lines, with a pointer to
207  * the header line in "b_linep".
208  * Buffers may be "Inactive" which means the files accosiated with them
209  * have not been read in yet. These get read in at "use buffer" time.
210  */
211 typedef struct BUFFER {
212  struct BUFFER *b_bufp; /* Link to next BUFFER */
213  struct LINE *b_dotp; /* Link to "." LINE structure */
214  struct LINE *b_markp; /* The same as the above two, */
215  struct LINE *b_linep; /* Link to the header LINE */
216  int b_doto; /* Offset of "." in above LINE */
217  int b_marko; /* but for the "mark" */
218  char b_active; /* window activated flag */
219  char b_nwnd; /* Count of windows on buffer */
220  char b_flag; /* Flags */
221  char b_mode; /* editor mode of this buffer */
222  char b_fname[NFILEN]; /* File name */
223  char b_bname[NBUFN]; /* Buffer name */
224 } BUFFER;

249 /*
250  * All text is kept in circularly linked lists of "LINE" structures. These
251  * begin at the header line (which is the blank line beyond the end of the
252  * buffer). This line is pointed to by the "BUFFER". Each line contains a the
253  * number of bytes in the line (the "used" size), the size of the text array,
254  * and the text. The end of line is not stored as a byte; it's implied. Future
255  * additions will include update hints, and a list of marks into the line.
256  */
257 typedef struct LINE {
258  struct LINE *l_fp; /* Link to the next line */
259  struct LINE *l_bp; /* Link to the previous line */
260  int l_size; /* Allocated size */
261  int l_used; /* Used size */
262  char l_text[1]; /* A bunch of characters. */
263 } LINE;
