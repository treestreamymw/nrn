1641 static PyObject* iternext_sl(PyHocObject* po, hoc_Item* ql) {
1642  // Note that the longstanding behavior of changing the currently
1643  // accessed section during iteration no longer takes place because
1644  // we cannot guarantee that an iterate will complete with state
1645  // PyHoc::Last and so the previous Section would have been left on the
1646  // hoc section stack.
1647 
1648  // Primarily the Section is pushed and the currently accessed python
1649  // Section is returned during sequential iteration over the list ql.
1650  // On re-entry here, the previous Section is popped.
1651  // The complexity is due to the possibility that the returned nrn_Section
1652  // may be deleted with h.delete_section(sec=nrn_Section). This would
1653  // invalidate the po->iteritem_ (point to freed memory) if it were
1654  // the iteritem of the current Section. Thus we choose to store
1655  // the next iteritem pointer in the list. Although not 100% safe, since
1656  // the user body of the iterator is allowed to delete_section an arbitary
1657  // subset of sections, the obvious work around, making a copy of ql,
1658  // is considered not worth it.
1659 
1660  // In this implementation, the first call to internext_sl starts out
1661  // in state PyHoc::Begin with po->iteritem_ == ql. If there is no valid
1662  // item, it sets po->iteritem_ = NULL and returns NULL. If there is
1663  // a valid item, it moves to state PyHoc::Last or PyHoc::NextNotLast
1664  // depending on whether there is a valid secitem following the current
1665  // item. The current section is pushed and currently accessed python
1666  // section is returned.
1667 
1668  // Thereafter, re-entry with po->iteritem_ == NULL, immediately
1669  // returns NULL.
1670 
1671  // Re-entry in state PyHoc::NextNotLast, pops the previously pushed Section,
1672  // sets sec to the current Section from po->iteritem_, and
1673  // sets po->iteritem_ to the next_valid_section. If there is no next_valid
1674  // section, then move to state PyHoc::Last.
1675  // Push the current section and return currently accessed python Section.
1676 
1677  // Re-entry in state PyHoc::Last just pops the current section, sets
1678  // po->iteritem_ = NULL, and returns NULL.
1679 
1680  if (po->iteritem_ == NULL) {
1681  return NULL;
1682  }
1683 
1684  if (po->u.its_ == PyHoc::Begin) {
1685  assert(po->iteritem_ == ql);
1686  hoc_Item* curitem = next_valid_secitem((hoc_Item*)(po->iteritem_), ql);
1687  if (curitem != ql) { // typical case, a valid current item
1688  Section* sec = curitem->element.sec;
1689  assert(sec->prop);
1690  // sec could be delete_section before return to internext.sl
1691  // which would invalidate curitem.
1692  // Not perfectly safe but leave po->iteritem_ as next valid
1693  // secitem after curitem.
1694  po->iteritem_ = next_valid_secitem(curitem, ql);
1695  if (po->iteritem_ == ql) {
1696  po->u.its_ = PyHoc::Last;
1697  }else{
1698  po->u.its_ = PyHoc::NextNotLast;
1699  }
1700  return (PyObject*)newpysechelp(sec);
1701  }else{ // no valid current item so stop
1702  po->iteritem_ = NULL;
1703  return NULL;
1704  }
1705  } else if (po->u.its_ == PyHoc::NextNotLast) {
1706  // it would be a bug if po->iteritem_ (now the curitem) has been delete_section
1707  Section* sec = ((hoc_Item*)(po->iteritem_))->element.sec;
1708  if (!sec->prop) {
1709  // Handle an edge case where sec is a Python
1710  // Section and happens to have had as its only reference the
1711  // iteration variable on the previous iteration.
1712  // i.e. when sec is returned from the previous iteration, it will
1713  // be referenced by the iteration variable and the previous reference
1714  // will go to 0, thus invalidating po->iteritem_->element.sec->prop
1715  po->iteritem_ = next_valid_secitem((hoc_Item*)(po->iteritem_), ql);
1716  if (po->iteritem_ == ql) {
1717  po->u.its_ = PyHoc::Last;
1718  po->iteritem_ = NULL;
1719  return NULL;
1720  }else{
1721  sec = ((hoc_Item*)(po->iteritem_))->element.sec;
1722  }
1723  }
1724  assert(sec->prop);
1725  po->iteritem_ = next_valid_secitem((hoc_Item*)(po->iteritem_), ql);
1726  if (po->iteritem_ == ql) {
1727  po->u.its_ = PyHoc::Last;
1728  }
1729  return (PyObject*)newpysechelp(sec);
1730  } else if (po->u.its_ == PyHoc::Last) {
1731  po->iteritem_ = NULL;
1732  return NULL;
1733  }
1734  return NULL; // never get here as po->u.its_ is always a defined state.
1735 }
