148 not, in the past, been reffed or unreffed.
149 The first problem is easily solved without much efficiency loss
150 by having the stack store the object pointer instead of the objref pointer.

287 static void frame_objauto_recover_on_err(Frame* ff) { /* only on error */
288  Frame* f;
289  for (f = fp; f > ff; --f) {
290  int i;
291  Symbol* sp = f->sp;
292  /* argn is the nargs argument on the stack. Stack items come in pairs
293  so stack increments are always multiples of 2.
294  Here, stkp is the last+1 localobj slot pair on the stack.
295  */
296  Datum* stkp = f->argn + 2 + sp->u.u_proc->nauto*2;
297  for (i = sp->u.u_proc->nobjauto; i > 0; --i) {
298  Object* ob = stkp[-2*i].obj;
299  hoc_obj_unref(ob);
300  /* Note that these AUTOOBJECT stack locations have an itemtype that
301  are left over from the previous stack usage of that location.
302  Regardless of that itemtype (e.g. OBJECTTMP), these did NOT
303  increment tobj_count so we need to guarantee that the subsequent
304  stack_obtmp_recover_on_err does not inadvertently free it again
305  by setting the itemtype to a non OBJECTTMP value. I hope this is
306  the only place where stack space was used in which no item type
307  was specified.
308  We are doing this here which happens rarely to avoid having to
309  set them when the stack obj pointers are zeroed.
310  */
311  stkp[-2*i + 1].i = 0;
312  }
