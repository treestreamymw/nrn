160 /*
161  * Write a character to the virtual screen. The virtual row and column are
162  * updated. If the line is too long put a "$" in the last column. This routine
163  * only puts printing characters into the virtual terminal buffers. Only
164  * column overflow is checked.

168 {

174  vtcol = (vtcol + 0x07) & ~0x07;
175  vp->v_text[term.t_ncol - 1] = '$';
176  } else if (c == '\t')
177  {
178  do
179  {
180  vtputc(' ');
181  }
182 #if TRUETAB
183  while ((vtcol%tabsize) != 0);
184 #else
185  while ((vtcol&0x07) != 0);
186 #endif
187  }

189  {
190  vtputc('^');
191  vtputc(c ^ 0x40);
192  }
193  else
194  vp->v_text[vtcol++] = c;

561  * row and column variables. It does try an exploit erase to end of line. The
562  * RAINBOW version of this routine uses fast video.
