147 rl_add_funmap_entry (name, function)
148  char *name;
149  Function *function;
150 {
151  if (funmap_entry + 2 >= funmap_size)
152  if (!funmap)
153  funmap = (FUNMAP **)xmalloc ((funmap_size = 80) * sizeof (FUNMAP *));
154  else
155  funmap =
156  (FUNMAP **)xrealloc (funmap, (funmap_size += 80) * sizeof (FUNMAP *));
157 
158  funmap[funmap_entry] = (FUNMAP *)xmalloc (sizeof (FUNMAP));
159  funmap[funmap_entry]->name = name;
160  funmap[funmap_entry]->function = function;
161 
162  funmap[++funmap_entry] = (FUNMAP *)NULL;
163 }

190 /* Produce a NULL terminated array of known function names. The array
191  is sorted. The array itself is allocated, but not the strings inside.
192  You should free () the array when you done, but not the pointrs. */
193 char **
194 rl_funmap_names ()
195 {
196  char **result = (char **)NULL;
197  int result_size, result_index;
198 
199  result_size = result_index = 0;
200 
201  /* Make sure that the function map has been initialized. */
202  rl_initialize_funmap ();
203 
204  for (result_index = 0; funmap[result_index]; result_index++)
205  {
206  if (result_index + 2 > result_size)
207  {
208  if (!result)
209  result = (char **)xmalloc ((result_size = 20) * sizeof (char *));
210  else
211  result = (char **)
212  xrealloc (result, (result_size += 20) * sizeof (char *));
213  }
214 
215  result[result_index] = funmap[result_index]->name;
216  result[result_index + 1] = (char *)NULL;
217  }
218 
219  qsort (result, result_index, sizeof (char *), qsort_string_compare);
220  return (result);
221 }
