81 void nrn_loc_point_process(int pointtype, Point_process* pnt, Section* sec, Node* node)
82 {
83  extern Prop *prop_alloc_disallow(), *prop_alloc();
84  extern Section* nrn_pnt_sec_for_need_;
85  Prop* p;
86  double x, nrn_arc_position();
87 
88  assert(!nrn_is_artificial_[pointsym[pointtype]->subtype]);
89  x = nrn_arc_position(sec, node);
90  /* the problem the next fragment overcomes is that POINTER's become
91  invalid when a point process is moved (dparam and param were
92  allocated but then param was freed and replaced by old param) The
93  error that I saw, then, was when a dparam pointed to a param --
94  useful to give default value to a POINTER and then the param was
95  immediately freed. This was the tip of the iceberg since in general
96  when one moves a point process, some pointers are valid and
97  some invalid and this can only be known by the model in its
98  CONSTRUCTOR. Therefore, instead of copying the old param to
99  the new param (and therefore invalidating other pointers as in
100  menus) we flag the allocation routine for the model to
101  1) not allocate param and dparam, 2) don't fill param but
102  do the work for dparam (fill pointers for ions),
103  3) execute the constructor normally.
104  */
105  if (pnt->prop) {
106  nrn_point_prop_ = pnt->prop;
107  }else{
108  nrn_point_prop_ = (Prop*)0;
109  }
110  nrn_pnt_sec_for_need_ = sec;
111  if (x == 0. || x == 1.) {
112  p = prop_alloc_disallow(&(node->prop), pointsym[pointtype]->subtype, node);
113  }else{
114  p = prop_alloc(&(node->prop), pointsym[pointtype]->subtype, node);
115  }
116  nrn_pnt_sec_for_need_ = (Section*)0;
117 
118  nrn_point_prop_ = (Prop*)0;
119  if (pnt->prop) {
120  pnt->prop->param = (double*)0;
121  pnt->prop->dparam = (Datum*)0;
122  free_one_point(pnt);
123  }
124  nrn_sec_ref(&pnt->sec, sec);
125  pnt->node = node;
126  pnt->prop = p;
127  pnt->prop->dparam[0].pval = &NODEAREA(node);
128  pnt->prop->dparam[1]._pvoid = (void*)pnt;
129  if (pnt->ob) {
130  if (pnt->ob->observers) {
131  hoc_obj_notify(pnt->ob);
132  }
