63  /*
64  treesolver
65 
66  a - above the diagonal
67  d - diagonal

70  pindex - parent indices
71  n - number of states
72  */

74  int i;
75 
76  /* triang */
77  for (i = n - 1; i > 0; --i) {
78  int pin = pindex[i];
79  if (pin > -1) {
80  double p;
81  p = a[i] / d[i];
82  d[pin] -= p * b[i];
83  rhs[pin] -= p * rhs[i];
