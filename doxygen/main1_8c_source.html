488 /*
489  * This is the general command execution routine. It handles the fake binding
490  * of all the keys to "self-insert". It also clears out the "thisflag" word,
491  * and arranges to move it to the "lastflag", so that the next command can
492  * look at it. Return the status of command.
493  */
494 int execute(c, f, n)
495  int c, f, n;
496 {
497  register KEYTAB *ktp;
498  register int status;
499 
500  ktp = &keytab[0]; /* Look in key table. */
501  while (ktp->k_fp != NULL) {
502  if (ktp->k_code == c) {
503  thisflag = 0;
504  status = (*ktp->k_fp)(f, n);
505  lastflag = thisflag;
506  return (status);
507  }
508  ++ktp;
509  }
510 
511  /*
512  * If a space was typed, fill column is defined, the argument is non-
513  * negative, wrap mode is enabled, and we are now past fill column,
514  * and we are not read-only, perform word wrap.
515  */
516  if (c == ' ' && (curwp->w_bufp->b_mode & MDWRAP) && fillcol > 0 &&
517  n >= 0 && getccol(FALSE) > fillcol &&
518  (curwp->w_bufp->b_mode & MDVIEW) == FALSE)
519  {IGNORE(wrapword(0));}
520 
521  if ((c>=0x20 && c<=0x7E) /* Self inserting. */
522  || (c>=0xA0 && c<=0xFE)) {
523  if (n <= 0) { /* Fenceposts. */
524  lastflag = 0;
525  return (n<0 ? FALSE : TRUE);
526  }
527  thisflag = 0; /* For the future. */
528 
529  /* if we are in overwrite mode, not at eol,
530  and next char is not a tab or we are at a tab stop,
531  delete a char forword */
532  if (curwp->w_bufp->b_mode & MDOVER &&
533  curwp->w_doto < curwp->w_dotp->l_used &&
534  (lgetc(curwp->w_dotp, curwp->w_doto) != '\t' ||
535  (curwp->w_doto) % tabsize == tabsize-1))
536  {IGNORE(ldelete(1, FALSE));}
537 
538  /* do the appropriate insertion */
539  if (c == '}' && (curbp->b_mode & MDCMOD) != 0)
540  status = insbrace(n, c);
541  else if (c == '#' && (curbp->b_mode & MDCMOD) != 0)
542  status = inspound();
543  else
544  status = linsert(n, c);
545 
546  lastflag = thisflag;
547  return (status);
548  }
549  mlwrite("\007[Key not bound]"); /* complain */
550  lastflag = 0; /* Fake last flags. */
551  return (FALSE);
552 }

556  * Do the standard keyboard preprocessing. Convert the keys to the internal
557  * character set.

559 int getkey()
560 {
561  int c;
562 #if AMIGA
563  int d;
564 #endif
565 
566  c = (*term.t_getchar)();
567 
568 #if RAINBOW
569 
570  if (c & Function_Key)
571  {
572  int i;
573 
574  for (i = 0; i < lk_map_size; i++)
575  if (c == lk_map[i][0])
576  return lk_map[i][1];
577  }
578  else if (c == Shift + 015) return CTRL | 'J';
579  else if (c == Shift + 0x7F) return META | 0x7F;
580 #endif
581 
582 #if MSDOS
583  if (c == 0) { /* Apply SPEC prefix */
584  c = getkey();
585  return(SPEC | c);
586  }
587 #endif
588 
589 #if AMIGA
590  /* apply SPEC prefix */
591  if ((unsigned)c == 155) {
592  c = (*term.t_getchar)();
593 
594  /* first try to see if it is a cursor key */
595  if ((c >= 'A' && c <= 'D') || c == 'S' || c == 'T')
596  return(SPEC | c);
597 
598  /* next, a 2 char sequence */
599  d = (*term.t_getchar)();
600  if (d == '~')
601  return(SPEC | c);
602 
603  /* decode a 3 char sequence */
604  c = d + 32;
605  /* if a shifted function key, eat the tilde */
606  if (d >= '0' && d <= '9')
607  d = (*term.t_getchar)();
608  return(SPEC | c);
609  }
610 #endif
611 
612  if (c == METACH) { /* Apply M- prefix */
613 #if MLH
614  mlwrite("M-:");
615  c = getctl();
616  mlerase();
617 #else
618  c = getctl();
619 #endif /*MLH*/
620  return (META | c);
621  }
622 
623  if (c>=0x00 && c<=0x1F) /* C0 control -> C- */
624  c = CTRL | (c+'@');
625  return (c);
626 }
