67  if (den == 0.0) x.error ("Attempted division by zero.");

179 Complex /* const */ sqrt(const Complex& x)
180 {
181  if (x.real() == 0.0 && x.imag() == 0.0)
182  return Complex(0.0, 0.0);
183  else
184  {
185  double s = sqrt((myfabs(x.real()) + hypot(x.real(), x.imag())) * 0.5);
186  double d = (x.imag() / s) * 0.5;
187  if (x.real() > 0.0)
188  return Complex(s, d);
189  else if (x.imag() >= 0.0)
190  return Complex(d, s);
191  else
192  return Complex(-d, -s);
193  }
194 }
