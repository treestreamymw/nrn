113 static uint32_t nrn_uint32hash( uint32_t a) {
114  a = (a ^ 61) ^ (a >> 16);
115  a = a + (a << 3);
116  a = a ^ (a >> 4);
117  a = a * 0x27d4eb2d;
118  a = a ^ (a >> 15);
119  return a;
120 }

136 NrnHash::NrnHash(int n) { \
137  for (size_ = 32; size_ < n; size_ <<= 1); \
138  first_ = new NrnHashEntry(NrnHash)*[size_]; \
139  --size_; \
140  last_ = &first_[size_]; \
141  for (register NrnHashEntry(NrnHash)** e = first_; e <= last_; e++) { \
142  *e = NULL; \
143  } \
144  nclash_ = nfind_ = 0; \
145 } \
