324 void geometry3d_vi(double* p1, double* p2, double v1, double v2, double* out) {
325  if (fabs(v1) < 0.000000000001) {
326  out[0] = p1[0]; out[1] = p1[1]; out[2] = p1[2];
327  return;
328  }
329  if (fabs(v2) < 0.000000000001) {
330  out[0] = p2[0]; out[1] = p2[1]; out[2] = p2[2];
331  return;
332  }
333  double delta_v = v1 - v2;
334  if (fabs(delta_v) < 0.0000000001) {
335  out[0] = p1[0]; out[1] = p1[1]; out[2] = p1[2];
336  return;
337  }
338  double mu = v1 / delta_v;
339  if (isnan(mu)) {
340  printf("geometry3d_vi error. delta_v = %g, v1 = %g, v2 = %g\n", delta_v, v1, v2);
341  }
342  out[0] = p1[0] + mu * (p2[0] - p1[0]);
343  out[1] = p1[1] + mu * (p2[1] - p1[1]);
344  out[2] = p1[2] + mu * (p2[2] - p1[2]);
345 }

459  double nx, ny, nz, y, yy, xx, x;
460  nx = px - cx;
461  ny = py - cy;
462  nz = pz - cz;
463  y = fabs(axisx * nx + axisy * ny + axisz * nz);
464  yy = y * y;
465  xx = nx * nx + ny * ny + nz * nz - yy;
466  if (y < h) {
467  return max(-fabs(h - y), sqrt(xx) - r);
468  } else {
469  y -= h;

471  if (xx < rr) {

473  } else {
474  x = sqrt(xx) - r;
475  return sqrt(yy + x * x);
476  }
477  }

530  nx = px - x0;
531  ny = py - y0;
532  nz = pz - z0;
533  y = axisx * nx + axisy * ny + axisz * nz;
534  yy = y * y;
535  xx = nx * nx + ny * ny + nz * nz - yy;
536  // in principle, xx >= 0, but roundoff errors may cause trouble
537  if (xx < 0) xx = 0;
538  if (y < 0) {
539  if (xx < rra) return -y;
540  x = sqrt(xx) - r0;
541  return sqrt(x * x + yy);
542  } else if (xx < rrb) {
543  return y - axislength;
544  } else {
545  x = sqrt(xx) - r0;
546  if (y < 0) {
547  if (x < 0) return y;
548  return sqrt(x * x + yy);
549  } else {
550  ry = x * side1 + y * side2;
551  if (ry < 0) return sqrt(x * x + yy);
552  rx = x * side2 - y * side1;
553  if (ry > conelength) {
554  ry -= conelength;
555  return sqrt(rx * rx + ry * ry);
556  } else {
