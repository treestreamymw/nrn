487  Prop* p;
488  int conc, rev;
489 {
490  int oldconc, oldrev;
491  int* it = &p->dparam[0].i;
492  oldconc = (*it & 03);
493  oldrev = (*it & 030)>>3;
494  /* precedence */
495  if (oldconc < conc) {
496  oldconc = conc;
497  }
498  if (oldrev < rev) {
499  oldrev = rev;
500  }
501  /* promote type */
502  if (oldconc > 0 && oldrev < 2) {
503  oldrev = 2;
504  }

506  *it += oldconc + 010*oldrev;
507  if (oldconc == 3) { /* if state then cinit */
508  *it += 4;

510  *it += 0100;
511  }
512  }
513  if (oldconc > 0 && oldrev == 2) { /*einit*/
514  *it += 040;
