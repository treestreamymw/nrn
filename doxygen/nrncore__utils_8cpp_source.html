121 // This function is related to stdindex2ptr in CoreNeuron to determine which values should
122 // be transferred from CoreNeuron. Types correspond to the value to be transferred based on
123 // mech_type enum or non-artificial cell mechanisms.
124 // Limited to pointers to voltage, nt._nrn_fast_imem->_nrn_sav_rhs (fast_imem value) or
125 // data of non-artificial cell mechanisms.
126 // Requires cache_efficient mode.
127 // Input double* and NrnThread. Output type and index.
128 // type == 0 means could not determine index.
129 int nrn_dblpntr2nrncore(double* pd, NrnThread& nt, int& type, int& index) {
130  assert(use_cachevec);
131  int nnode = nt.end;
132  type = 0;
133  if (pd >= nt._actual_v && pd < (nt._actual_v + nnode)) {
134  type = voltage; // signifies an index into voltage array portion of _data
135  index = pd - nt._actual_v;
136  } else if (nt._nrn_fast_imem && pd >= nt._nrn_fast_imem->_nrn_sav_rhs && pd < (nt._nrn_fast_imem->_nrn_sav_rhs + nnode)) {
137  type = i_membrane_; // signifies an index into i_membrane_ array portion of _data
138  index = pd - nt._nrn_fast_imem->_nrn_sav_rhs;
139  }else{
140  for (NrnThreadMembList* tml = nt.tml; tml; tml = tml->next) {
141  if (nrn_is_artificial_[tml->index]) { continue; }
142  Memb_list* ml1 = tml->ml;
143  int nn = nrn_prop_param_size_[tml->index] * ml1->nodecount;
144  if (pd >= ml1->data[0] && pd < (ml1->data[0] + nn)) {
145  type = tml->index;
146  index = pd - ml1->data[0];
147  break;
148  }
149  }
150  }
151  return type == 0 ? 1 : 0;
152 }
