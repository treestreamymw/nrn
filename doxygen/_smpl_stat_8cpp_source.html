58 // t-distribution: given p-value and degrees of freedom, return t-value
59 // adapted from Peizer & Pratt JASA, vol63, p1416
60 
61 double tval(double p, int df)
62 {
63  double t;
64  int positive = p >= 0.5;
65  p = (positive)? 1.0 - p : p;
66  if (p <= 0.0 || df <= 0)
67  t = HUGE_VAL;
68  else if (p == 0.5)
69  t = 0.0;
70  else if (df == 1)
71  t = 1.0 / tan((p + p) * 1.57079633);
72  else if (df == 2)
73  t = sqrt(1.0 / ((p + p) * (1.0 - p)) - 2.0);
74  else
75  {
76  double ddf = df;
77  double a = sqrt(log(1.0 / (p * p)));
78  double aa = a * a;
79  a = a - ((2.515517 + (0.802853 * a) + (0.010328 * aa)) /
80  (1.0 + (1.432788 * a) + (0.189269 * aa) +
81  (0.001308 * aa * a)));
82  t = ddf - 0.666666667 + 1.0 / (10.0 * ddf);
83  t = sqrt(ddf * (exp(a * a * (ddf - 0.833333333) / (t * t)) - 1.0));
84  }
85  return (positive)? t : -t;
86 }
