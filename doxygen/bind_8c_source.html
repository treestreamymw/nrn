89 int cmdstr(c, seq) /* change a key command to a string we can print out */
90 
91 int c; /* sequence to translate */
92 char *seq; /* destination string for sequence */
93 
94 {
95  char *ptr; /* pointer into current position in sequence */
96 
97  ptr = seq;
98 
99  /* apply meta sequence if needed */
100  if (c & META) {
101  *ptr++ = 'M';
102  *ptr++ = '-';
103  }
104 
105  /* apply ^X sequence if needed */
106  if (c & CTLX) {
107  *ptr++ = '^';
108  *ptr++ = 'X';
109  }
110 
111  /* apply SPEC sequence if needed */
112  if (c & SPEC) {
113  *ptr++ = 'F';
114  *ptr++ = 'N';
115  }
116 
117  /* apply control sequence if needed */
118  if (c & CTRL) {
119  *ptr++ = '^';
120  }
121 
122  c = c & 255; /* strip the prefixes */
123 
124  /* and output the final sequence */
125 
126  *ptr++ = c;
127  *ptr = 0; /* terminate the string */
128  return TRUE;

477 /* docmd: take a passed string as a command line and translate
478  it to be executed as a command. This function will be
479  used by execute-command-line and by all source and
480  startup files.
481 
482  format of the command line is:
483 
484  {# arg} <command-name> {<argument string(s)>}
485 */
486 
487 int docmd(cline)
488 
489 char *cline; /* command line to execute */
490 
491 {
492  register char *cp; /* pointer to current position in command */
493  register char *tp; /* pointer to current position in token */
494  register int f; /* default argument flag */
495  register int n; /* numeric repeat value */
496  register int sign; /* sign of numeric argument */
497  register int (*fnc)(); /* function to execute */
498  register int status; /* return status of function */
499  register int oldcle; /* old contents of clexec flag */
500  char token[NSTRING]; /* next token off of command line */
501  int (*fncmatch())();
502  char *gettok();
503 
504  /* first set up the default command values */
505  f = FALSE;
506  n = 1;
507 
508  cp = cline; /* start at the begining of the line */
509  cp = gettok(cp, token); /* and grab the first token */
510 
511  /* check for and process numeric leadin argument */
512  if ((token[0] >= '0' && token[0] <= '9') || token[0] == '-') {
513  f = TRUE;
514  n = 0;
515  tp = &token[0];
516 
517  /* check for a sign! */
518  sign = 1;
519  if (*tp == '-') {
520  ++tp;
521  sign = -1;
522  }
523 
524  /* calc up the digits in the token string */
525  while(*tp) {
526  if (*tp >= '0' && *tp <= '9')
527  n = n * 10 + *tp - '0';
528  ++tp;
529  }
530  n *= sign; /* adjust for the sign */
531 
532  /* and now get the command to execute */
533  cp = gettok(cp, token); /* grab the next token */
534  }
535 
536  /* and match the token to see if it exists */
537  if ((fnc = fncmatch(token)) == NULL) {
538  mlwrite("[No such Function]");
539  return(FALSE);
540  }
541 
542  /* save the arguments and go execute the command */
543  Strcpy(sarg, cp); /* save the rest */
544  oldcle = clexec; /* save old clexec flag */
545  clexec = TRUE; /* in cline execution */
546  status = (*fnc)(f, n); /* call the function */
547  clexec = oldcle; /* restore clexec flag */
548  return(status);
549 }

556 
557 char *src, *tok; /* source string, destination token */
558

561  while (*src == ' ' || *src == '\t')

566  ++src; /* past the quote */

599 int getckey() /* get a command key sequence from the keyboard */
600 
601 {
602  register int c; /* character fetched */
603  register char *tp; /* pointer into the token */
604  char tok[NSTRING]; /* command incoming */
605 
606  /* check to see if we are executing a command line */
607  if (clexec) {
608  IGNORE(nxtarg(tok)); /* get the next token */
609 
610  /* parse it up */
611  tp = &tok[0];
612  c = 0;
613 
614  /* first, the META prefix */
615  if (*tp == 'M' && *(tp+1) == '-') {
616  c = META;
617  tp += 2;
618  }
619 
620  /* next the function prefix */
621  if (*tp == 'F' && *(tp+1) == 'N') {
622  c |= SPEC;
623  tp += 2;
624  }
625 
626  /* control-x as well... */
627  if (*tp == '^' && *(tp+1) == 'X') {
628  c |= CTLX;
629  tp += 2;
630  }
631 
632  /* a control char? */
633  if (*tp == '^' && *(tp+1) != 0) {
634  c |= CTRL;
635  ++tp;
636  }
637 
638  /* make sure we are not lower case */
639  if (c >= 'a' && c <= 'z')
640  c -= 32;
641 
642  /* the final sequence... */
643  c |= *tp;
644 
645  return(c);
646  }
647 
648  /* or the normal way */
649  c = getkey(); /* get a command sequence */
650  if (c == (CTRL|'X')) /* get control-x sequence */
651  c = CTLX | getctl();
652  return(c);
653 }
