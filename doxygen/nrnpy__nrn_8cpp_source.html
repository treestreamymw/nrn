1357 static PyObject* seg_volume(NPySegObj* self) {
1358  Section* sec = self->pysec_->sec_;
1359  int i;
1360  if (sec->recalc_area_) {
1361  nrn_area_ri(sec);
1362  }
1363  double x = self->x_;
1364  double a = 0.0;
1365  // volume only exists on the interior
1366  if (x > 0. && x < 1.) {
1367  // every section owns one extra node (the conservation node)
1368  int nseg = sec->nnode - 1;
1369  double length = section_length(sec) / nseg;
1370  int iseg = x * nseg;
1371  double seg_left_arc = iseg * length;
1372  double seg_right_arc = (iseg + 1) * length;
1373  if (sec->npt3d > 1) {
1374  int i_left = arg_bisect_arc3d(sec, sec->npt3d, seg_left_arc);
1375  double left_diam = fabs(sec->pt3d[i_left].d);
1376  double right_diam = fabs(sec->pt3d[i_left + 1].d);
1377  double left_arc = seg_left_arc;
1378  left_diam = interpolate(sec->pt3d[i_left].arc, sec->pt3d[i_left + 1].arc, left_diam, right_diam, left_arc);
1379 
1380  for (i = i_left + 1; i < sec->npt3d && sec->pt3d[i].arc < seg_right_arc; i++) {
1381  right_diam = fabs(sec->pt3d[i].d);
1382  a += scaled_frustum_volume(sec->pt3d[i].arc - left_arc, left_diam, right_diam);
1383  left_arc = sec->pt3d[i].arc;
1384  left_diam = right_diam;
1385  }
1386  if (i < sec->npt3d) {
1387  right_diam = interpolate(left_arc, sec->pt3d[i].arc, left_diam, fabs(sec->pt3d[i].d), seg_right_arc);
1388  a += scaled_frustum_volume(seg_right_arc - left_arc, left_diam, right_diam);
1389  }
1390 
1391  // the scaled_frustum_volume formula factored out a pi/12
1392  a *= M_PI / 12;
1393  } else {
1394  // 0 or 1 3D points... so give cylinder volume
1395  Node* nd = node_exact(sec, x);
1396  for (Prop* p = nd->prop; p; p = p->next) {
1397  if (p->type == MORPHOLOGY) {
1398  double diam = p->param[0];
1399  a = M_PI * diam * diam / 4 * length;
1400  break;
1401  }
1402  }
1403  }
1404  }
1405  PyObject* result = Py_BuildValue("d", a);
1406  return result;
1407 }
